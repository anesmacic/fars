def MapKeys2States2Regiona(state,stateDictionary,regionDictionary,regionIndex):
    state = stateDictionary[state]
    for _,region in regionDictionary.items():
        if (state in region):
            for key,values in regionDictionary.items():
                if region==values:
                    return regionIndex[key]

from geojson import Feature, FeatureCollection, Point
def populate_geojson(dataframe):
    features = []
    dataframe.apply(lambda x: (
                    features.append(
            Feature(
                geometry = Point((x['LONGITUD'], x['LATITUDE'])),
                properties = {
                    'persons': x['PERSONS'],
                    'region': x['REGION']
                }
            )
        ) 
        # neater plotting for non-composite geoprojections
        if (x['LONGITUD'] < 0.0 and x['LONGITUD'] > -125.0) else None
        )
                    ,axis=1)
    collection = FeatureCollection(features)
    with open("GeoObs.geojson", "w") as f:
        f.write('%s' % collection)

def populate_response_time(data):
    response_time = data.apply(lambda x: 
                                        calculate_response_time(
                                            x
                                        ) 
     
                               if time_valid(x) else time_valid_sub(x)
                                        ,
                                        axis=1)
    data['RESPONSE_TIME'] = response_time
    
def calculate_response_time(row):
    HOSP_HR, NOT_HOUR, HOSP_MIN, NOT_MIN = row[['HOSP_HR','NOT_HOUR','HOSP_MN','NOT_MIN']]
    
    if HOSP_HR < NOT_HOUR:
        hour_diff = HOSP_HR+23-NOT_HOUR # interval crossing midnight

    elif HOSP_HR == NOT_HOUR:
        hour_diff = 0
    else:
        hour_diff = HOSP_HR - NOT_HOUR -1
        
    if hour_diff >= 0:
        min_diff = HOSP_MIN + (60-NOT_MIN)
    else:
        min_diff = HOSP_MIN - NOT_MIN

    result = hour_diff*60+min_diff 
    
    return result if result < 180 else 180


def time_valid(row):
    
    if row['HOSP_HR'] > 23 or row['NOT_HOUR'] > 23 or row['HOSP_MN'] > 59 or row['NOT_MIN'] > 59:
        return False
    return True

def time_valid_sub(row):
    if row['HOSP_HR'] < 23 or row['NOT_HOUR'] < 23:
        if row['HOSP_MN'] > 59 or row['NOT_MIN'] > 59:
            return -1
        else:
            return (row['HOSP_HR'] - row['NOT_HOUR'])*60
    else:
        return -1

def plotMRTbyregion(tt,region):
    t = tt[tt['REGION']==region]
    t = t[t['RESPONSE_TIME']>0]
  
    if region == 0:
        color = "red"
        name = "Northeast"
    if region == 1:
        color = "green"
        name = "Midwest"
    if region == 2:
        color = "blue"
        name= "West"
    if region == 3:
        color = "orange"
        name = "South"

    sb.distplot(t['RESPONSE_TIME'], bins=50, kde=True, color=color,hist=False,label=(name))

def silhouette_chart(sample_silhouette_values, n_clusters):
    y_lower = 10
   
    for i in range(n_clusters):
        # Aggregate the silhouette scores for samples belonging to
        # cluster i, and sort them
        ith_cluster_silhouette_values = sample_silhouette_values[cluster_labels == i]
        ith_cluster_silhouette_values.sort()

        size_cluster_i = ith_cluster_silhouette_values.shape[0]
        y_upper = y_lower + size_cluster_i

        color = cm.nipy_spectral(float(i) / n_clusters)
        plt.fill_betweenx(np.arange(y_lower, y_upper), 0, ith_cluster_silhouette_values,
                          facecolor=color, edgecolor=color, alpha=0.7)

        # Label the silhouette plots with their cluster numbers at the middle
        plt.text(-0.05, y_lower + 0.5 * size_cluster_i, str(i))

        # Compute the new y_lower for next plot
        y_lower = y_upper + 10  # 10 for the 0 samples
        
    plt.title("The silhouette plot for the various clusters.")
    plt.xlabel("The silhouette coefficient values")
    plt.ylabel("Cluster label")

    # The vertical line for average silhouette score of all the values
    plt.axvline(x=silhouette_avg, color="red", linestyle="--")
    plt.show()
